106    def __init__(self, variables: List[Variable]=None, coefficients: List[float]=None):
107        if variables is None:
108            self.variables = []
109            self.coefficients = []
110        else:
111            self.variables = variables
112            self.coefficients = coefficients

114    def add_term(self, variable: Variable, coefficient: float) -> None:
115        """
116        Adds a term to the linear expression.
117
118        Args:
119            variable (Variable): The variable of the term.
120            coefficient (float): The coefficient of the term.
121        """
122        if variable in self.variables:
123            self.coefficients[self.variables.index(variable)] += coefficient
124        else:
125            self.variables.append(variable)
126            self.coefficients.append(coefficient)
127
128    def __add__(self, other: "LinearExpr") -> "LinearExpr":
129        """
130        Adds two linear expressions.
131
132        Args:
133            other (LinearExpr): The other linear expression.
134
135        Returns:
136            LinearExpr: The sum of the two linear expressions.
137        """
138        result = LinearExpr(self.variables, self.coefficients)
139
140        if isinstance(other, LinearExpr):
141            for variable, coefficient in zip(self.variables, self.coefficients):
142                result.add_term(variable, coefficient)
143        elif isinstance(other, (int, float)):
144            result.add_term(Variable("constant", 0, 0), other)
145        else:
146            raise TypeError("Unsupported operand type(s) for +: 'LinearExpr' and '{}'".format(type(other)))
147
148        return result
149
150    def __sub__(self, other: "LinearExpr") -> "LinearExpr":
151        """
152        Subtracts two linear expressions.
153
154        Args:
155            other (LinearExpr): The other linear expression.
156
157        Returns:
158            LinearExpr: The difference of the two linear expressions.
159        """
160        return self + (-other)

162    def __neg__(self) -> "LinearExpr":
163        """
164        Negates a linear expression.
165
166        Returns:
167            LinearExpr: The negated linear expression.
168        """
169
170        result = LinearExpr()
171        for variable, coefficient in zip(self.variables, self.coefficients):
172            result[variable] = -coefficient
173        return result

175    def __mul__(self, scalar: Union[int,float]) -> "LinearExpr":
176        """
177        Performs scalar multiplication on a linear expression.

179        Args:

183        if not isinstance(scalar, (int, float)):
